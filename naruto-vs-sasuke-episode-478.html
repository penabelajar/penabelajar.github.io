[tex]( {10}^{3} \div {10}^{2} \times 100) \: \times \: ( {10}^{2} \times 3 \div 4) \\ = 1000 \times 75 \\ = 75000[/tex]

[tex] {10}^{3} - {25}^{2} \\ = 1000 - 625 \\ = 375[/tex]

1 hari=

[tex] {10}^{3 - 2} \times {10}^{2} [/tex]

[tex]10 ^{1 + 2} = {10}^{3} = 1000 \: mangkuk[/tex]

[tex]10^2 × 3 ÷ 4 [/tex]

[tex]100 \times 3 \div 4[/tex]
