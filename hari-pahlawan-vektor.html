a = (1, -4, 3)
b = (2, 5, -1)
Q = (4, 1, 6)

PQ = 2a - 3b
q - p = 2a - 3b
p = q - (2a - 3b)
P = q - 2a + 3b
p = (4, 1, 6) - 2(1, -4, 3) + 3(2, 5, -1)
p = (4, 1, 6) - (2, -8, 6) + (6, 15, -3)
